const int fixedFee = 2;

void init()
{}

void run()
{
    int price;
    price = addFee(100); // The price is going to be 100$ + fee
}

int addFee (int amount)
{
    int total = amount + fixedFee;
    turnOnLight(); // Turn on the register light when done processing

    return total;
}

void turnOnLight ()
{
    Q#100.00 = true;
}
